Summarrising data (mean and variance)

fls1 <- fls_raw %>% 
  filter(study != 23) %>% 
  filter(study != 27) %>%
  filter(study != 38) %>%
  filter(study != 48) %>%
  group_by(study, year, location, state, cultivar, planting_date, app_date, growth_stage, brand_name, group, active_ingred,n) %>% 
  summarise(mean_sev = mean(sev),
            mean_yld = mean(yld))
## `summarise()` has grouped output by 'study', 'year', 'location', 'state', 'cultivar',
## 'planting_date', 'app_date', 'growth_stage', 'brand_name', 'group', 'active_ingred'. You can
## override using the `.groups` argument.
fls2_sev <- fls_raw %>% 
  filter(study != 23) %>% 
  filter(study != 27) %>% 
  filter(study != 38) %>%
  filter(study != 48) %>%
  filter(sev>0) %>% 
  group_by(study, year) %>%  
  select(active_ingred, rep, sev) %>%
  group_by(study, year) %>% 
  do(tidy(aov(.$sev ~ .$active_ingred + factor(.$rep)))) %>% 
  filter(term == "Residuals") %>% 
  select(1,2,6) %>% 
  set_names("study", "year", "v_sev")
## Adding missing grouping variables: `study`, `year`
fls2_yld <- fls_raw %>% 
  filter(study != 23) %>% 
  filter(study != 27) %>%
  filter(study != 38) %>%
  filter(study != 48) %>%
  group_by(study, year) %>%  
  select(active_ingred, rep, yld) %>% 
  group_by(study, year) %>% 
  do(tidy(aov(.$yld ~ .$active_ingred + factor(.$rep)))) %>% 
  filter(term == "Residuals") %>% 
  select(1,2,6) %>% 
  set_names("study", "year", "v_yld")
## Adding missing grouping variables: `study`, `year`
qmr = left_join(fls2_sev, fls2_yld)
## Joining, by = c("study", "year")
fls_trial = full_join(fls1, qmr)
